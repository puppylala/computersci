from decimal import *  
list1=[]  
def nansile(a):  
    b = 1  
    sum=0  
    d=100  
    e=0  
    while abs(d-e)>0.000001:  
        c=(a+b**2)/(2*b)  
        #jilu  
        d=b  
        e=c  
        b=c  
        sum+=1  
    e=Decimal(str(e)).quantize(Decimal("0."+"1"*2),rounding=ROUND_HALF_UP)  
    list1.append(f"{sum} {e}")  
while True:  
    try:  
        a=float(input())  
        nansile(a)  
    except EOFError:  
        break  
for i in list1:  
    print(i)
