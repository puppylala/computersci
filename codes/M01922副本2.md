```python
import math  
while True:  
        a=int(input())  
        if a==0:  
            break  
        list1=[]  
        for i in range(a):  
            b,c=map(int,input().split())  
            if c<0:  
                continue  
            t=(4500*3.6+b*c)/b  
            list1.append(t)  
        list1.sort()  
        a=math.ceil(list1[0])  
        if list1[0]-a>=0.5:  
            print(a+1)  
        else:  
            print(a)
```
