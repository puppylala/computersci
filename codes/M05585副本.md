```python
a=int(input())  
list1=[]  
for i in range(a):  
    sumr=0  
    sumb=0  
    b=int(input())  
    matrix=[[] for j in range(b)]  
    visit=[[False for m in range(b)] for k in range(b)]  
    for i in range(b):  
        matrix[i]=list(input())  
    def diamond(i,j,type):  
        if i<0 or j<0 or i>b-1 or j>b-1 or visit[i][j]:  
            return  
        if matrix[i][j]!=type or visit[i][j]:  
            return  
        visit[i][j] = True  
        diamond(i-1,j,type)  
        diamond(i+1,j,type)  
        diamond(i,j-1,type)  
        diamond(i,j+1,type)  
    for p in range(b):  
        for q in range(b):  
            if matrix[p][q]=="#":  
                continue  
            if matrix[p][q]=="b" and visit[p][q]==False:  
                sumb+=1  
                diamond(p,q,"b")  
            if matrix[p][q]=="r" and visit[p][q]==False:  
                sumr+=1  
                diamond(p,q,"r")  
    list1.append(sumr)  
    list1.append(sumb)  
for i in range(a):  
    print(f"{list1[2*i]} {list1[2*i+1]}")
```
