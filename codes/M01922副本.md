```python
import math  
a=int(input())  
list1=[]  
for i in range(a):  
    b,c=map(int,input().split(" "))  
    if c<0:  
        continue  
    t=(4500+b*c/3.6)/(b/3.6)  
    list1.append(t)  
list1.sort()  
print(list1[0])  
a=math.ceil(list1[0])  
if list1[0]-a>=0.5:  
    print(a+1)  
else:  
    print(a)
```
