```python
""" 大概思路是这样 提议应该是每一行都有或者每一列都有，这两个条件满足一个就可以  
那么我要分别讨论满足一行的情况与满足一列的情况  
还要让一个坐标匹配 这个要和行与列结合起来才可以  
最终分类是这样：因为要保证第二条必定有一个坐标等于编号的行或列，首先找出一种这样的情况，然后根据这个继续下去"""  
a=int(input())  
c=0  
d=0  
list1=[]  
for i in range(a):  
    b=int(input())  
    lie=list(map(int,input().split()))  
    hang=list(map(int,input().split()))#保证每一列都有取到 此时满足了第二条件  
    hang1=sorted(hang)  
    c=hang1[0]*len(lie)+sum(lie[:])  
    lie1=sorted(lie)  
    d=lie1[0]*len(lie)+sum(hang[:])  
    list1.append(min(c,d))  
for m in list1:  
    print(m)
```
