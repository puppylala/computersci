```python
a=int(input())  
list1=[]  
for i in range(a):  
    list1.append(int(input()))  
for i in range(len(list1)):  
    for j in range(0,100):  
        if 2**j<list1[i]:  
            continue  
        elif 2**j>list1[i]:  
            list1[i]=(1+list1[i])*list1[i]/2-2*(2**j-1)  
            break  
        elif 2**j==list1[i]:  
            list1[i]=(1+list1[i])*list1[i]/2-2*(2**(j+1)-1)  
            break  
for i in range(len(list1)):  
    print(int(list1[i]))
```
