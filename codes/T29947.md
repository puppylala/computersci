```python
a,b=map(int,input().split())  
list1=[]  
for i in range(b):  
    c=list(map(int,input().split()))  
    list1.append(c)  
list1.sort()  
la=0  
while b==1:  
    print(a-list1[0][1]+list1[0][0])  
    quit()  
while b==0:  
    print(a+1)  
for i in range(b-1):  
    if list1[i][1]>=list1[i+1][1]:  
        list1[i+1]=list1[i]  
    elif list1[i][1]>=list1[i+1][0]:  
        list1[i+1][0]=list1[i][0]  
    else:  
        la+=list1[i+1][0]-list1[i][1]-1  
        list1[i+1][0]=list1[i][0]  
print(a-(list1[-1][1]-list1[-1][0])+la)
```
