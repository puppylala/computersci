```python
a,b=input().split()  
a,b=int(a),int(b)  
list1=list(map(int,input().split()))  
dict1={}  
list2=[]  
def a_c(list3:list,c:int):  
    global b  
    if c==1:  
        for k in range(len(list3)):  
            list2.append(list3[k])  
            dict1[tuple(sorted(list2))]=0  
            del list2[-1]  
    else:  
        for i in range(len(list3)):  
            list2.append(list3[i])  
            a_c(list3[0:i]+list3[i+1:],c-1)  
            del list2[-1]  
a_c(list1,b)  
list4=dict1.keys()  
list4=sorted(list4)  
for i in range(len(list4)):  
    print(" ".join(map(str,list4[i])))
```
