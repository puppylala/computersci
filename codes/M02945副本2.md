def missiles(k, heights):  
    dp = [1] * k  
    for i in range(1, k):  
        for j in range(i):  
            if heights[i] <= heights[j]:  
                dp[i]=max(dp[i], dp[j] + 1)  
    return max(dp)  
k=int(input())  
heights=list(map(int, input().split()))  
result=missiles(k, heights)  
print(result)  
#这动态规划也太神奇了我去 它可以记录每一个节点的最长不递增序列然后直接用上一次用的就可以了 真厉害
