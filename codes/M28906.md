            list3=list2.copy()

            list3.append(n-k)  
            list1.append(list3)  
  
    elif  max(list2)>=int((n+1)/2+1):  
        pass  
    else:  
        for k in range(max(list2),n+1):  
            list2.append(k)  
            nub(n-k,m-1)
