```python
a=int(input())  
dict1={}  
list1=[]  
b=list(map(int,input().split()))  
bb=sorted(b)  
time=0  
for i in range(a):  
    time+=(a-1-i)*bb[i]  
for i in range(a):  
    if b[i] in dict1.keys():  
        dict1[b[i]+0.0001*(i+1)]=i+1  
        b[i]+=0.0001*(i+1)  
    else:  
        dict1[b[i]]=i+1  
b.sort()  
for i in range(a):  
    list1.append(dict1[b[i]])  
print(" ".join(map(str,list1)))  
print(f"{time/a:.2f}")
```
