```python
ROMAN={"I":1,"V":5,"X":10,"L":50,"C":100,"D":500,"M":1000}  
INT={a:b for b,a in ROMAN.items()}  
c=input()  
list1=[]  
if ord(c[0])>=48 and ord(c[0])<=57:  
    #整数  
    c=int(c)  
    d=6  
    while c>0:  
        count=c//list(ROMAN.values())[d]  
        list1.append(INT[list(ROMAN.values())[d]]*count)  
        c-=count*list(ROMAN.values())[d]  
        d-=1  
    aaa="".join(list1)  
    aaa=aaa.replace("DCCCC","CM")  
    aaa=aaa.replace("CCCC","CD")  
    aaa=aaa.replace("LXXXX","XC")  
    aaa=aaa.replace("XXXX","XL")  
    aaa=aaa.replace("VIIII","IX")  
    aaa=aaa.replace("IIII","IV")  
    print(aaa)  
else:  
    aaa=0  
    for i in range(len(c)):  
        if i==0:  
            aaa+=ROMAN[c[i]]  
        elif ROMAN[c[i]]>ROMAN[c[i-1]]:  
           aaa+=ROMAN[c[i]]-ROMAN[c[i-1]]*2  
        else:  
            aaa+=ROMAN[c[i]]  
    print(aaa)
```
