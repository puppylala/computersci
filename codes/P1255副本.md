```python
import sys  
sys.setrecursionlimit(1000000)  # 设置递归深度限制为100万  
a=int(input())  
dict1={}  
sum1=0  
sum2=0  
def floors(b:int):  
    global dict1,sum1,sum2  
    if b==1 or b==0:  
        sum1+=1  
        sum2+=1  
    elif b in dict1.keys():  
        sum1+=dict1[b]  
        sum2+=dict1[b]  
    else:  
        floors(b-1)  
        floors(b-2)  
        dict1[b]=sum2  
floors(a)  
print(sum1)  
#成功
```
