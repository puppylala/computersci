#要找一个最长降序  
a=int(input())  
list1=list(map(int,input().split()))  
list5=[]  
def bullet(list3:list):  
    if not list3:  
        return  
    tik=list3[0]  
    sum1=0  
    list4=[0]  
    for i in list3:  
        if i<=tik:  
            sum1+=1  
            tik=i  
        else:  
            list4.append(sum1)  
            sum1=0  
            tik=list3[0]  
            if i<=tik:  
                sum1+=1  
                tik=i  
            else:  
                continue  
    list5.append(max(list4)+list3.count(list3[0])-1)  
    del list3[0]  
    bullet(list3)  
bullet(list1)  
print(list5)  
print(max(list5))
