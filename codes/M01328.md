list1=[]  
def greedy(f,g,h):  
    mmm=(h**2-g**2)**0.5+f  
    return round(mmm,2)  
while True:  
    found=True  
    mm=input()  
    if mm=="":  
        continue  
    else:  
        a,b=map(int,mm.split())  
    if a==0 and b==0:  
        break  
    list2=[]  
    for i in range(a):  
        (c,d)=map(int,input().split())  
        list2.append((c,d))  
    for (c,d) in list2:  
        if d>b or b<=0:  
            list1.append(-1)  
            found=False  
            break    if found==True:  
        m=0  
        n=0  
        list2.sort()  
        while list2!=[]:  
            m+=1  
            k=greedy(list2[0][0],list2[0][1],b)  
            for i in range(len(list2)):  
                if (list2[i][0]-k)**2+list2[i][1]**2>b**2:  
                    n=i  
            list2=list2[n:]  
        list1.append(m)  
for i in range(len(list1)):  
    print(f"Case {i+1}: {list1[i]}")
