```python
a,b,c=map(int,input().split(","))  
dict1={}  
def nubsum(aa:int):  
    bb=0  
    for k in range(len(str(aa))):  
        bb+=int(str(aa)[k])  
    return bb  
for i in range(a+1,b):  
    if nubsum(int(i))%c==0:  
        if nubsum(i) not in dict1:  
            dict1[nubsum(i)]=[]  
            dict1[nubsum(i)].append(str(i))  
        else:  
            dict1[nubsum(i)].append(str(i))  
list1=sorted(dict1.keys())  
for i in list1:  
    print(",".join(dict1[i]))
```
