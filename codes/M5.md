```python
a=input()  
list1=[]  
for i in range(len(a)):  
    list1.append(a[-1-i])  
b="".join(list1)  
tik=0  
dict1={}  
def repeat(c:str):#这里用的时反义的B  
    global a,tik  
    try:  
        d=c.index(a[tik])  
        if len(a)-d-tik==1 or len(a)-d-tik==0:  
            dd=c.index(a[0])  
            dict1[a[0:len(a)-dd+1]]=len(a)-dd  
            tik=0  
        elif a[tik+1]==c[d+1]:  
            tik+=1  
            repeat(c[d+1:])  
        else:  
            pass  
    except IndexError:  
        pass  
for i in range(len(a)):  
    a=a[i:]  
    repeat(b)  
dict1=dict(sorted(dict1.items(),key=lambda x:x[1]))  
print(list(dict1.keys())[0])
```
