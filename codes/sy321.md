```python
"""这是一个BFS 非常关键的两点：  
1 BFS使用队列数据结构，每一次都从后面加入队列，然后踢出第一个。  
2 BFS不用计数变量来记录数据或者是行走的路径，而是将这些信息装进元组里和位置信息一起打包！"""  
rows,cols=map(int,input().split())  
graph=[[] for i in range(rows)]  
visited = [[False for k in range(cols)] for j in range(rows)]  
for i in range(rows):  
    graph[i]=list(map(int,input().split()))  
for i in range(rows):  
    for j in range(cols):  
        if graph[i][j]==1:  
            visited[i][j]=True  
list1=[]  
def bfs():  
    global visited,list1  
    step=0  
    start=(0,0,[(0,0)],step)  
    queue=[start]  
    direction=[(0,1),(0,-1),(-1,0),(1,0)]  
    while queue:  
        now=queue.pop(0)  
        visited[now[0]][now[1]]=True  
        if now[0]==rows-1 and now[1]==cols-1:  
            list1.append((now[2],now[3]))  
        else:  
            for i in direction:  
                x,y=now[0]+i[0],now[1]+i[1]  
                if 0<=x<rows and 0<=y<cols and not visited[x][y]:  
                    queue.append((x,y,now[2]+[(x,y)],now[3]+1))  
bfs()  
list1.sort(key=lambda x:x[1],reverse=True)  
for i in list1[0][0]:  
    print(f"{i[0]+1} {i[1]+1}")
```
