```python
a=int(input())  
b=input()  
b=list(b)  
for i in range(len(b)):  
    i=int(i)  
    if ord(b[i])>=65 and ord(b[i])<=90:  
        while ord(b[i])-a<65:  
            b[i]=chr(ord(b[i])+26)  
    if ord(b[i])>=97 and ord(b[i])<=122:  
        while ord(b[i])-a<97:  
            b[i]=chr(ord(b[i])+26)  
for i in range(len(b)):  
    b[i]=chr(ord(b[i])-a)  
print("".join(b))
```
