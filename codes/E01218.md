import sys  
sys.setrecursionlimit(1<<30)  
aa=int(input())  
list1=[10,2]  
for n in range(1,aa+1):  
    a=int(input())  
    list=[0]*a  
    bb=0  
    for i in range(1,a+1):  
        for j in range(i,a+1):  
            if j%i==0:  
                if list[j-1]==0:  
                    list[j-1]=1  
                else:list[j-1]=0  
            else:continue  
    for k in range(0,a):  
        if list[k]==1:  
            bb+=1  
    list1[n-1]=bb  
print(list1[0])  
print(list1[1])
