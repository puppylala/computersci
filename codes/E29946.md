```python
a=input()  
b=len(a)  
k=int(input())  
c=b-k#要留的位数  
d=c  
list1=[]  
list2=[]  
for i in range(len(a)):  
    list1.append(a[i])  
if c<=0:  
    print(0)  
    quit()  
for i in range(d):  
    if i==0:  
        list3=list(filter(lambda x:x!="0",list1))  
        aa = min(list3[0:len(list3) - c + 1])  
        aaa = list3.index(aa)  
        list2.append(aa)  
        list1 = list1[aaa + 1:]  
        c -= 1  
        continue  
    aa=min(list1[0:len(list1)-c+1])  
    aaa=list1.index(aa)  
    list2.append(aa)  
    list1=list1[aaa+1:]  
    c-=1  
print(int("".join(map(str,list2))))
```
