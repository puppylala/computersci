c=0  
d=0  
e=0  
f=input()  
for i in range(1,len(f),2):  
    try:  
        if int(f[i-1])==1 and int(f[i])==0:  
            e+=1  
    except(ValueError):  
        if int(f[i-1])==1:  
            c+=1  
        elif int(f[i-1])==5:  
            d+=1  
        elif int(f[i-1])==10:  
            e+=1  
print(c,d,e)
