a=int(input())  
for i in range(1,a+1):  
    found=False  
    for j in range(i,a+1):  
        if found:  
            break  
        else:  
            for m in range(1,j+1):  
                if found:  
                    break  
                else:  
                    for k in range(1,m+1):  
                        if i**3==m**3+j**3+k**3:  
                            print(f"Cube = {i}, Triple = ({j},{m},{k})")  
                            found=True  
                            break
