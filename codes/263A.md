```python
list1=[0]*5  
list2=[0]*5  
a=0  
for i in range(0,5):  
    list1[i]=input().split()  
for k in range(0,5):  
    list2[k]=[list1[l][k] for l in range(0,5)]  
for j in range(0,5):  
    if "1" in list1[j]:  
        if list1[j].index("1")==2:  
            break  
        else:  
            a+=abs(list1[j].index("1")-2)  
    else:continue  
for m in range(0,5):  
    if "1" in list2[m]:  
        if list2[m].index("1")==2:  
            break  
        else:  
            a+=abs(list2[m].index("1")-2)  
    else:continue  
print(a)
```
