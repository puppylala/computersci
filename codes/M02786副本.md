```python
a = int(input())  
list1 = []  
for i in range(a):  
    b = int(input())  
    if b == 1:  
        list1.append(1)  
    elif b == 2:  
        list1.append(2)  
    else:  
        pre2=1  
        pre1=2  
        current=0  
        for j in range(3, b + 1):  
            current=(2*pre1+pre2)%32767  
            #滚动更新  
            pre2,pre1=pre1,current  
        list1.append(current)  
for i in list1:  
    print(i)  
#滚动更新就不超时了 前提是current要提前取模
```
