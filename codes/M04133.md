```python
rang=int(input())  
spot=int(input())  
rub={}  
max=0  
sum=1  
for i in range(spot):  
    a,b,c=map(int,input().split())  
    if (a,b) not in rub:  
        rub[(a,b)] = c  
    else:  
        rub[(a,b)]+=c  
#前缀和  
x=[[0 for k in range(0,1026)] for i in range(0,1026)]  
#这里大了一圈  
for i in range(1,1025):  
    for j in range(1,1025):  
        weight=rub.get((i-1,j-1),0)  
        x[i][j]=weight+x[i-1][j]+x[i][j-1]-x[i-1][j-1]  
for i in range(1025):  
    for j in range(1025):  
        if x[i+1][j+1]-x[i-2*rang][j+1]-x[i+1][j-2*rang]+x[i-2*rang][j-2*rang]>max:  
            max=x[i+1][j+1]  
            sum=1  
        elif x[i+1][j+1]-x[i-2*rang][j+1]-x[i+1][j-2*rang]+x[i-2*rang][j-2*rang]==max:  
            sum+=1  
print(f"{sum} {max}")
```
