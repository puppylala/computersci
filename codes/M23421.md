```python
a,b=map(int,input().split())  
list1=list(map(int,input().split()))#价格  
list2=list(map(int,input().split()))#重量  
list3=[]#单价  
dict1={}  
sum1=0  
list4=[]  
for i in range(a):  
    list3.append(list1[i]/list2[i])  
for i in range(a):  
    dict1[list3[i]]=(list1[i],list2[i])  
list3.sort(reverse=True)  
for k in range(a):  
    if not list3:  
        break  
    for i in range(len(list3)):  
        if b-dict1[list3[i]][1]>=0:  
            sum1+=dict1[list3[i]][0]  
            b-=dict1[list3[i]][1]  
        else:  
             continue  
    list4.append(sum1)  
    sum1=0  
    del list3[0]  
print(max(list4))
```
