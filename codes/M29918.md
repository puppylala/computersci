```python
a=int(input())  
dict1={i:-i for i in range(1,a+1)}  
for i in range(1,int(a**0.5)+1):  
    dict1[i*i]+=i  
    for j in range(i,a//i+1):  
            dict1[i*j]+=(i+j)  
for i in range(2,a+1):  
    j=dict1[i]  
    if i<j<=a and dict1[j]==i:  
        print(f"{i} {j}")  
#最终结果为188ms
```
