def goods(list5,list6,e):  
    global sum1,list4  
    if min(list6)>e:  
        list4.append(sum1)  
    for k in range(len(list5)):  
        if list6[k]<=e:  
            sum1+=list5[k]  
            goods(list5[0:k]+list5[k+1:],list6[0:k]+list6[k+1:],e-list6[k])  
            sum1-=list5[k]
