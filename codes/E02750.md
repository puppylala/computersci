```python
a=int(input())  
b=1  
found=True  
if a%2!=0:  
    print(0,0)  
else:  
    while found:  
        if (a-b*4)==0:  
            print(b,b*2)  
            found=False  
        elif (a-b*4)==2:  
            print(b+1,b*2+1)  
            found=False  
        else:  
            b+=1
```
