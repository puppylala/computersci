found=True  
list1=[]  
list2=[]  
list3=[]  
while found:  
    a,b=map(int,input().split())  
    if a==0 and b==0:  
        found=False  
        break    list1.append(a)  
    list2.append(b)  
for i in range(len(list1)):  
    list4=[j for j in range(1,list1[i]+1)]  
    a=list2[i]  
    while len(list4)>1:  
        while a>len(list4):  
            a=a-len(list4)  
        del list4[a-1]  
        a+=list2[i]  
        a-=1  
    list3.append(list4[0])  
for i in list3:  
    print(i)
