```python
import sys  
a=input()  
list=[]  
for i in range(len(a)):  
    if ord(a[i])>=49 and ord(a[i])<=57:  
        list.append(a[i])  
    if  ord(a[i])==48:  
        if  ord(a[i-1])>=48 and ord(a[i-1])<=57:  
            list.append(a[i])  
        else:continue  
    if ord(a[i])<=47 or ord(a[i])>=58:  
        if ord(a[i-1])>=48 and ord(a[i-1])<=57:  
            list.append(" ")  
        else:continue  
for i in range(len(list)):  
    if list[i]== " ":  
        print("")  
    if list[i]!= " ":  
        sys.stdout.write(list[i])
```
