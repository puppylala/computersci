```python
a=int(input())  
list1=[]  
list2=[]  
for i in range(a):  
    b=int(input())  
    listr=[]  
    listb=[]  
    matrix=[[] for k in range(b+1)]  
    sumr=0  
    sumb=0  
    for i in range(b):  
        dd=input()  
        for j in range(b):  
            matrix[i].append(dd[j])  
    for i in range(b):  
        matrix[i].append("#")  
    matrix[b]=["#"*(b+1)]  
    def diamond(i,j):  
        global list2  
        if j==b and i==b:  
            return 0  
        elif j==b:  
            if list2!=[]:  
                c=list2[0]  
                list2=[]  
                diamond(i+1,c)  
            else:  
                diamond(i+1,0)  
        elif matrix[i][j]=="#":  
            diamond(i,j+1)  
        elif matrix[i][j]=="r":  
            list2.append(j)  
            listr.append(1)  
            matrix[i][j]="#"  
            diamond(i,j+1)  
        elif matrix[i][j]=="b":  
            list2.append(j)  
            listb.append(1)  
            matrix[i][j]="#"  
            diamond(i,j+1)  
    diamond(0, 0)  
    for i in range(len(listr)-1):  
        if listr[i]==1:  
            if listr[i+1]==0 or i+1==len(listr):  
                sumr+=1  
        elif i+1==len(listr) and listr[i+1]==1:  
            sumr+=1  
        elif i+1==len(listr):  
            break  
    for i in range(len(listb)-1):  
        if listb[i]==1:  
            if listb[i+1]==0 or i+1==len(listb):  
                sumb+=1  
        elif i+1==len(listb) and listb[i+1]==1:  
            sumb+=1  
        elif i+1==len(listb):  
            break  
    list1.append(sumr)  
    list1.append(sumb)  
for i in range(a):  
    print(f"{list1[2*i]} {list1[2*i+1]}")
```
