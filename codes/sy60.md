a,b=input().split()  
a,b=map(int,[a,b])  
c=0  
list1=[]  
def aaa(c):  
    if int(c[0])**3+int(c[1])**3+int(c[2])**3==int(c):  
        return True  
    else:  
        return False  
for i in range(a,b+1):  
    if aaa(str(i)):  
        list1.append(i)  
        c=1  
if c==0:  
    print("NO")  
else:  
    print(' '.join(map(str,list1)))
