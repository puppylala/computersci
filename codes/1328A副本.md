list1=[]  
def aaa(a,e):  
    aa=0  
    found=True  
    while found:  
        if a%e==0:  
            found=False  
            list1.append(aa)  
        else:  
            f=(a//e+1)*e  
            aa=f-a  
            list1.append(aa)  
            break  
b=int(input())  
for i in range(0,b):  
    c,d=input().split()  
    aaa(int(c),int(d))  
for i in range(0,b):  
    print(list1[i])
