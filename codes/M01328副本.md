```python
results = []  
while True:  
    line = input()  
    if not line:  
        continue  
    n, d = map(int, line.split())  
    if n == 0 and d == 0:  
        break  
    islands = []  
    found=True  
    for i in range(n):  
        x, y = map(int, input().split())  
        islands.append((x, y))  
        if y > d:  
            found=False  
    if not found or d < 0:  
        results.append(-1)  
        continue  
    intervals = []  
    for x, y in islands:  
        dx =(d*d-y*y)**0.5  
        intervals.append((x-dx,x+dx))  
    intervals.sort(key=lambda x: x[1])  
    radar_count = 0  
    position=intervals[0][0]-1  
    for left, right in intervals:  
        if left>position:  
            radar_count+=1  
            position= right  
    results.append(radar_count)  
for i, count in enumerate(results, 1):  
    print(f"Case {i}: {count}")
```
