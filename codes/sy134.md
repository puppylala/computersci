list1=list(map(int,input().split()))  
list1.sort()  
dict1={}  
list4=[]  
def c_and_a(list3:list):  
    global dict1,list4  
    for k in list3:  
        if len(list3)==1:  
            list4.append(str(k))  
            list6=list4.copy()  
            dict1[tuple(list6)]=0  
        else:  
            list4.append(str(k))  
            list5=list3.copy()  
            list5.remove(k)  
            c_and_a(list5)  
        del list4[-1]  
c_and_a(list1)  
for i in dict1.keys():  
    print(" ".join(i))
