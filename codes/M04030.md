a=input()  
b=input()  
e=0  
for i in range(len(b)):  
    if b[i]==" ":  
        e+=1  
    else:  
        break  
b=b.strip()  
b=b.split()  
sum=0  
d=0  
if not a or not b:  
    print("-1")  
    exit()  
for i in range(len(b)):  
    if a.lower()==b[i] or a.capitalize()==b[i]:  
       sum+=1  
if sum>=1:  
    try:  
        c=b.index(a.capitalize())  
        if c==0:  
            print(f"{sum} {e}")  
        else:  
            for i in range(1,c+1):  
                d+=len(b[i-1])+1  
            print(f"{sum} {d+e}")  
    except ValueError:  
        c = b.index(a.lower())  
        if c == 0:  
            print(f"{sum} {e}")  
        else:  
            for i in range(1, c + 1):  
                d += len(b[i - 1]) + 1  
            print(f"{sum} {d+e}")  
else:  
    print("-1")
