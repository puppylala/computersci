```python
def solve():  
    data = input().split()  
    X = int(data[0])  
    coins = list(map(int,input().split()))  
    coins = [c for c in coins if c <= X]  
    if not coins:  
        if X == 0:  
            return 0  
        else:  
            return -1  
    coins.sort()  
    if coins[0] > 1:  
        return -1  
    max_reach = 0  
    count = 0  
    while max_reach < X:  
        candidate = -1  
        for coin in coins:  
            if coin <= max_reach + 1:  
                candidate = coin  
            else:  
                break  
        if candidate == -1:  
            return -1  
        max_reach += candidate  
        count += 1  
        if max_reach >= X:  
            break  
    return count  
print(solve())
```
