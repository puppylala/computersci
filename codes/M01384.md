INF=float('inf')  
def pigpig():  
    global sum1,weight,acceptable,money,c,INF  
    dp=[INF for m in range(acceptable+1)]  
    dp[0]=0  
    #这个DP我先用最笨的方法  
    for k in range(c):  
        for j in range(weight[k],acceptable+1):  
            dp[j]=min(dp[j],dp[j-weight[k]]+money[k])  
    return dp[-1]  
list1=[]

    c=int(input())

    weight=[]  
    for i in range(c):  
        aaa,bbb=input().split()

        bbb=int(bbb)  
        money.append(aaa)  
        weight.append(bbb)

    list1.append(pigpig())
