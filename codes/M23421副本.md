```python
a,b=input().split()  
a,b=int(a),int(b)  
list1=list(map(int,input().split()))#价格  
list2=list(map(int,input().split()))#重量  
def steal():  
    global list1,list2  
    dp=[0 for i in range(b+1)]#这个动态规划的关键：对于每一个物品都有放或者不放两种状态。通过比较两种状态的差异来决定放或者不放。首先遍历每一个物品，然后在遍历所有可以放进去的背包重量  
    for i in range(len(list1)):  
        for j in range(b,list2[i]-1,-1):  
            dp[j]=max(dp[j],dp[j-list2[i]]+list1[i])  
    return max(dp)  
print(steal())
```
