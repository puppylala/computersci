line,con=input().split()  
line=int(line)  
con=int(con)  
matrices1=[0]*line  
matrices2=[[] for i in range(con)]#如果真的[[]]*0的话会导致引用位置相同，造成超级大列表。  
SUM=0  
sum=0  
def judge(a):  
    global sum  
    if 1 in a:  
        b=a.index(1)  
        if b==len(a)-1:  
            sum+=1  
        elif a[b+1]==1:  
            del a[0:b+1]  
            judge(a)  
        else:  
            sum+=1  
            del a[0:b+2]  
            judge(a)  
    return sum*2  for i in range(line):  
    matrices1[i]=list(map(int,input().split()))  
for i in range(line):  
    for j in range(con):  
        matrices2[j].append(matrices1[i][j])  
for i in range(con):  
    SUM+=judge(matrices2[i])  
    sum=0  
for i in range(line):  
    SUM+=judge(matrices1[i])  
    sum=0  
print(SUM)
