```python
a,o=map(int,input().split())  
dict1={}  
f=0  
g=0  
for i in range(a):  
    b,c=map(int,input().split())  
    if b/c not in dict1.keys():  
        dict1[b/c]=[b,c]  
    else:  
        dict1[b/c][1]+=c  
d=sorted(dict1.keys(),reverse=True)  
while o!=0 and g<=len(d)-1:  
    f+=min(o,dict1[d[g]][1])*d[g]  
    o-=min(o,dict1[d[g]][1])  
    g+=1  
print(f"{f:.2f}")
```
