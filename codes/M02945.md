a=int(input())  
list1=list(map(int,input().split()))  
list5=list1.copy()  
sum1=0  
list4=[]  
max1=0  
def bullet(list2:list):  
    global max1,sum1,list4  
#这里我想要计算用每一个导弹高度做最大值的时候后面能拦截的个数  
    if len(list2)==1:  
        list4.append(1)  
    else:  
         list3=list2[::-1]  
         b=len(list2)-list3.index(list2[0])-1  
         sum1+=list2.count(list2[0])-1  
         list6=list5[0:b]  
         max1=list6[0]  
         for k in list6:  
             if k>max1:  
                 continue  
             else:  
                 max1=k  
                 sum1+=1  
         max1=list2[b]  
         for i in range(b,len(list2)):  
             if list2[i]>max1:  
                 continue  
             else:  
                 max1=list2[i]  
                 sum1+=1  
         list4.append(sum1)  
         sum1=0  
         del list2[0]  
         bullet(list2)  
bullet(list1)  
print(max(list4))
