```python
n = int(input())  
age={}  
aaa=0  
list=[]  
for i in range(n):  
    name,agee = input().split()  
    if int(agee)<60:  
        list.append(name)  
        continue  
    while (int(agee),aaa) in age:  
        aaa-=1  
    age[(int(agee),aaa)]=name  
    aaa=0  
sorted_age = sorted(age.keys(), reverse=True)  
for i in sorted_age:  
    print(age[i])  
for i in list:  
    print(i)
```
