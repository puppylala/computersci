    if e==2:  
        print(f"1:{aa}->{bb}")  
        print(f"2:{aa}->{cc}")

    if e>2:  
        aaaa(e-1,aa,cc,bb)  
        print(f"{e}:{aa}->{cc}")  
        aaaa(e-1,bb,aa,cc)  
    return 0
