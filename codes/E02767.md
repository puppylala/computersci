```python
list1=input().split()  
for i in range(len(list1)):  
    for j in range(1,len(list1[i])+1):  
        if ord(list1[i][j-1])>=70 and ord(list1[i][j-1])<=90:  
            list1[i]=list1[i][0:j-1]+chr(ord(list1[i][j-1])-5)+list1[i][j:]  
        elif ord(list1[i][j-1])>=65 and ord(list1[i][j-1])<=69:  
            list1[i]=list1[i][0:j-1]+chr(ord(list1[i][j-1])+21)+list1[i][j:]  
    print(list1[i],end=' ')
```
