```python
a=int(input())  
list1=[]  
count=0  
for i in range(a):  
    (b,c)=map(int,input().split())  
    list1.append((b,c))  
list1.sort(key=lambda x:x[0])  
left=list1[0][0]-list1[0][1]-1  
for i in range(len(list1)):  
    if list1[i][0]-list1[i][1]>left:  
        left=list1[i][0]  
        count+=1  
    elif i==len(list1)-1 or list1[i][0]+list1[i][1]<list1[i+1][0]:  
        left=list1[i][0]+list1[i][1]  
        count+=1  
    else:  
        left=list1[i][0]  
print(count)
```
