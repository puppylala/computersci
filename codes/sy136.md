```python
a,b=input().split()  
a=int(a)  
b=int(b)  
list1=list(map(int,input().split()))  
list1.sort()  
dict1={}  
list3=[]  
def a_and_c(list2:list,c:int):  
    for i in list2:  
        if c==1:  
            list3.append(i)#这里要先按照数字序排序然后再转为字符串  
            dict1[tuple(sorted(list3))]=0#这里元组是不可变的。之后为了节省copy可以使用元组。  
        else:  
            list3.append(i)  
            list4=list2.copy()  
            list4.remove(i)  
            a_and_c(list4,c-1)  
        list3.remove(list3[-1])  
a_and_c(list1,b)  
for k in dict1.keys():  
    print(" ".join(map(str,k)))
```
