```python
a=int(input())  
b=list(map(int,input().split()))  
b.sort()  
wuqi=0  
list1=[]  
list2=[]  
for i in b:  
    wuqi+=i  
    list2.append(wuqi)  
def ourweapons(aa):  
    if aa >=list2[-1]:  
        return len(list2)  
    for k in range(len(list2)):  
        if aa<list2[k]:  
            return k  
    return 0  
found=True  
ddd=0  
while found:  
    if ourweapons(a)==0:  
        print(0)  
        quit()  
    list1.append(ourweapons(a)-ddd)  
    a+=b[-1]  
    del b[-1],list2[-1]  
    ddd+=1  
    if ourweapons(a)==len(list2):  
        list1.append(ourweapons(a)-ddd)  
        found=False  
list1.sort()  
print(list1[-1])
```
